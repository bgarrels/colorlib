colorLib.webServices

 
 
 
 
Name
 
brightnessDiff ( )
 
Examples
 
  
import colorLib.*;

Swatch s1;
Swatch s2;
color  c;

void setup() {
	size(200, 200);
	noStroke();
		
	s1 = new Swatch(this, color(123, 2, 48));
	s2 = new Swatch(this, color(3, 78, 255));
	c  = color(35, 78, 199);
		
	fill(s1.getColor());
	rect(0, 0, width/2, height);
	fill(s2.getColor());
	rect(width/2, 0, width/2, height/2);
	fill(c);
	rect(width/2, height/2, width/2, height/2);
		
	println("Brightness Difference between s1 and s2: " + s1.brightnessDiff(s2));
	println("Brightness Difference between s1 and c: " + s1.brightnessDiff(c));
}
 
 
Description
 
Calculated the brightness difference between the swatch and the given color or swatch. The brightness is calculated by the following formular suggest by the W3C: ((Red value * 299) + (Green value * 587) + (Blue value * 114)) / 1000 The result should be higher then 125 for a good readability.
 
Syntax
 
  
brightnessDiff(i_color);
brightnessDiff(i_swatch);
 
 
Parameters
i_color
i_swatch
 
Returns
float, difference
